def insertion_sort(arr):

    for i in range(1, len(arr)):

        current = arr[i]
        j = i - 1
 
        while j >= 0 and arr[j] > current:
            arr[j+1] = arr[j]
            j -= 1
      
        arr[j+1] = current


arr = [5, 2, 8, 3, 9, 6]
insertion_sort(arr)
print(arr)  # RÃ©sultat : [2, 3, 5, 6, 8, 9]
